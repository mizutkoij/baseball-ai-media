## 🛠️ **技術実装詳細**

### **NPBチームデータベース (`lib/npb-teams.ts`)**
```typescript
export interface NPBTeam {
  code: string;           // チームコード
  name: string;           // 正式名称
  shortName: string;      // 短縮名
  league: 'central' | 'pacific';
  colors: {
    primary: string;      // メインカラー
    secondary: string;    // サブカラー
    accent?: string;      // アクセントカラー
  };
  location: string;       // 本拠地
  stadium: string;        // 本拠地球場
  founded: number;        // 創設年
}
```

### **カラー取得ヘルパー関数**
```typescript
export function getTeamColor(teamName: string, type: 'primary' | 'secondary' | 'accent' = 'primary'): string
export function getTeamByName(name: string): NPBTeam | undefined  
export function getTeamsByLeague(league: 'central' | 'pacific'): NPBTeam[]
```
